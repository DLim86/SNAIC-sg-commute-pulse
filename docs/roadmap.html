# Geocode the destination with OneMap
def geocode(address, token):
    r = requests.get(
        "https://www.onemap.gov.sg/api/common/elastic/search",
        params={"searchVal": address, "returnGeom": "Y", "getAddrDetails": "Y"},
        headers={"Authorization": token}
    )
    result = r.json()["results"][0]
    return float(result["LATITUDE"]), float(result["LONGITUDE"])

# Get OneMap token (run once, valid 3 days)
def get_onemap_token(email, password):
    r = requests.post("https://www.onemap.gov.sg/api/auth/post/getToken",
                      json={"email": email, "password": password})
    return r.json()["access_token"]

# scripts/ingest.py — Flatten OneMap route response
import pandas as pd
from datetime import datetime, timezone

def fetch_routes(origin_lat, origin_lng, dest_lat, dest_lng, token, event_id):
    r = requests.get(
        "https://www.onemap.gov.sg/api/public/routingsvc/route",
        params={
            "token": token,
            "start": f"{origin_lat},{origin_lng}",
            "end": f"{dest_lat},{dest_lng}",
            "routeType": "pt",
            "date": "06-23-2026",
            "time": "09:00:00",
            "mode": "TRANSIT",
            "numItineraries": 3
        }
    ).json()

    rows = []
    for i, it in enumerate(r.get("plan", {}).get("itineraries", [])):
        rows.append({
            "option_id": f"{event_id}_opt{i}",
            "event_id": event_id,
            "total_duration_min": it["duration"] // 60,
            "walk_distance_m": it["walkDistance"],
            "num_transfers": it["transfers"],
            "fare": it.get("fare", 0),
            "fetched_at": datetime.now(timezone.utc)
        })
    return pd.DataFrame(rows)

# scripts/schema.py — Run once to create all tables
import duckdb

con = duckdb.connect("db/commute.duckdb")

con.execute("""
CREATE TABLE IF NOT EXISTS calendar_events (
    event_id     VARCHAR PRIMARY KEY,
    title        VARCHAR,
    start_time   TIMESTAMPTZ,
    location_raw VARCHAR,
    dest_lat     DOUBLE,
    dest_lng     DOUBLE,
    ingested_at  TIMESTAMP DEFAULT now()
);""")

con.execute("""
CREATE TABLE IF NOT EXISTS route_options (
    option_id          VARCHAR PRIMARY KEY,
    event_id           VARCHAR REFERENCES calendar_events(event_id),
    total_duration_min INTEGER,
    walk_distance_m    INTEGER,
    num_transfers      INTEGER,
    fare               DECIMAL(4,2),
    fetched_at         TIMESTAMP
);""")

con.execute("""
CREATE TABLE IF NOT EXISTS weather_forecast (
    area        VARCHAR,
    forecast    VARCHAR,
    is_rainy    BOOLEAN,
    valid_start TIMESTAMPTZ,
    valid_end   TIMESTAMPTZ,
    fetched_at  TIMESTAMP,
    PRIMARY KEY (area, valid_start)
);""")

con.execute("""
CREATE TABLE IF NOT EXISTS bus_arrivals (
    bus_stop_code VARCHAR,
    service_no    VARCHAR,
    next_bus_mins INTEGER,
    load          VARCHAR,   -- SEA / SDA / LSD (seats)
    fetched_at    TIMESTAMP,
    PRIMARY KEY (bus_stop_code, service_no, fetched_at)
);""")

con.execute("""
CREATE TABLE IF NOT EXISTS train_alerts (
    alert_id     VARCHAR PRIMARY KEY,
    affected_line VARCHAR,
    message      VARCHAR,
    severity     VARCHAR,
    fetched_at   TIMESTAMP
);""")

con.execute("""
CREATE TABLE IF NOT EXISTS recommendations (
    event_id          VARCHAR PRIMARY KEY,

    total_duration_min INTEGER,
    leave_by          TIMESTAMPTZ,
    estimated_arrival TIMESTAMPTZ,
    weather_warning   VARCHAR,
    disruption_warning VARCHAR,
    reason            VARCHAR,
    created_at        TIMESTAMP DEFAULT now()
);""")

Data Transformation
SQL Cleaning & Recommendation Logic
Write SQL views that join your three data streams — routes, weather, and alerts — into a single enriched recommendation table. Calculate the leave-by time (event start minus travel duration minus a 10-minute buffer). Flag rainy walks and disrupted MRT lines. Score routes and pick the best one per event.
SQL JOINs CASE WHEN Window functions INTERVAL arithmetic CREATE VIEW 
-- scripts/transform.sql — Core recommendation view

CREATE OR REPLACE VIEW v_enriched_routes AS
SELECT
    r.option_id,
    r.event_id,
    r.total_duration_min,
    r.walk_distance_m,
    r.num_transfers,
    r.fare,
    e.start_time,
    e.title,
    e.dest_lat,
    e.dest_lng,

    -- When to leave (10 min buffer)
    e.start_time - INTERVAL (r.total_duration_min + 10) MINUTE AS leave_by,

    -- Closest weather area (simplified: use dest area)
    w.forecast AS weather_forecast,
    w.is_rainy,

    -- Any active train alerts?
    CASE WHEN ta.alert_id IS NOT NULL THEN ta.message ELSE NULL END AS alert_msg,

    -- Recommendation scoring
    CASE
        WHEN w.is_rainy AND r.walk_distance_m > 400
            THEN '⚠ Rainy — take covered transport'
        WHEN ta.alert_id IS NOT NULL
            THEN '⚠ MRT disruption — add 20 min buffer'
        WHEN r.total_duration_min = MIN(r.total_duration_min) OVER
                (PARTITION BY r.event_id)
            THEN '✓ Fastest option'
        ELSE 'Alternative route'
    END AS recommendation_reason,

    -- Rank routes per event (1 = best)
    ROW_NUMBER() OVER (
        PARTITION BY r.event_id
        ORDER BY
            CASE WHEN w.is_rainy AND r.walk_distance_m > 400 THEN 1 ELSE 0 END,
            r.total_duration_min
    ) AS route_rank

FROM route_options r
JOIN calendar_events e ON r.event_id = e.event_id
LEFT JOIN weather_forecast w
    ON w.fetched_at = (SELECT MAX(fetched_at) FROM weather_forecast)
LEFT JOIN train_alerts ta
    ON ta.severity = 'HEAVY'
   AND ta.fetched_at > NOW() - INTERVAL '30 minutes';
Key Transformation Decisions
Leave-by timeevent start − duration − 10 min buffer. The 10-min buffer is a design choice — document it.
Rain + walking ruleFlag walks over 400m as inadvisable when rain is forecast. Threshold is adjustable.
Disruption detectionJoin train_alerts only for rows within the last 30 minutes — stale alerts should not affect recommendations.
Route rankingWindow function ranks options per event. Rank 1 = recommended. Penalise rainy walks first, then sort by duration.

Streamlit Dashboard

Streamlit DuckDB Python API pandas st.metric st.dataframe 
# scripts/serve.py — Run with: streamlit run scripts/serve.py
import streamlit as st
import duckdb, pandas as pd
from datetime import datetime

st.set_page_config(page_title="SG Commute Recommender", page_icon="🚇", layout="wide")

@st.cache_data(ttl=300)  # refresh every 5 minutes
def load_recommendation():
    con = duckdb.connect("db/commute.duckdb", read_only=True)
    return con.execute("""
        SELECT * FROM v_enriched_routes
        WHERE route_rank = 1
        ORDER BY start_time
        LIMIT 1
    """).df()

def load_all_options():
    con = duckdb.connect("db/commute.duckdb", read_only=True)
    return con.execute("""
        SELECT total_duration_min, walk_distance_m, num_transfers,
               fare, recommendation_reason, route_rank
        FROM v_enriched_routes ORDER BY route_rank
    """).df()

st.title("🚇 SG Commute Recommender")
rec = load_recommendation()

if not rec.empty:
    r = rec.iloc[0]
    st.subheader(f"Next: {r['title']}")

    col1, col2, col3 = st.columns(3)
    col1.metric("Leave by", r["leave_by"].strftime("%H:%M"))
    col2.metric("Travel time", f"{r['total_duration_min']} min")
    col3.metric("Fare", f"S${r['fare']:.2f}")

    if r["is_rainy"]:
        st.warning("🌧 Rain forecast at destination — avoid long walks.")
    if r["alert_msg"]:
        st.error(f"🚨 {r['alert_msg']}")

    st.caption(r["recommendation_reason"])
    st.dataframe(load_all_options(), use_container_width=True)
